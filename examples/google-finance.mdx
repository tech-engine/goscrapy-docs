This example demonstrates a real-world advanced pattern: reverse-engineering Google Finance's internal batch RPC API to extract stock quotes, company info, chart data, and news.

**Key concepts demonstrated:**

- POST requests with custom headers and body
- Passing metadata between callbacks using `Meta()`
- Parsing non-HTML batch RPC responses using `gjson` and `gosm`
- Job-based spider triggering from `main.go`

## main.go

```go
func main() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    spider, err := google_finance.New(ctx)
    // ... error handling

    ticker := "GOOGL:NASDAQ"
    job := google_finance.NewJob(ticker)
    spider.StartRequest(ctx, job)

    if err := spider.Wait(true); err != nil && !errors.Is(err, context.Canceled) {
        fmt.Printf("Engine finished with error: %v\n", err)
    }
}
```

## spider.go

```go
func (s *Spider) StartRequest(ctx context.Context, job *Job) {
    ticker := job.Query()
    body, rpcids := buildBatchRequest(ticker)

    targetUrl := fmt.Sprintf("%s?rpcids=%s&source-path=/finance/quote/%s&hl=en&gl=us&rt=c",
        GOOGLE_RPC_URL, rpcids, ticker)

    req := s.Request(ctx).
        Url(targetUrl).
        Method("POST").
        Body(body).
        Meta("ticker", ticker)

    for k, v := range RPC_HEADERS {
        req.SetHeader(k, v)
    }

    s.Parse(req, s.parseResponse)
}

func (s *Spider) parseResponse(ctx context.Context, resp core.IResponseReader) {
    tickerRaw, _ := resp.Meta("ticker")
    ticker := tickerRaw.(string)

    results := parseBatchResults(resp.Bytes())
    record := &Record{J: NewJob(ticker)}
    mapQuote(ticker, results["xh8wxf"], record)
    mapCompany(results["HqGpWd"], record)
    mapChart(results["AiCwsd"], record)
    mapNews(results["nBEQBc"], record)

    s.Yield(record)
}
```
