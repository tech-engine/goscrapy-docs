This example scrapes [books.toscrape.com](https://books.toscrape.com), extracting book titles, prices, ratings, and descriptions across paginated listings.

**Key concepts demonstrated:**

- CSS selectors for pagination (`li.next a@href`)
- Gosm declarative mapping for product details
- Multi-callback spider pattern (listing → detail)

## Spider Logic

```go
func (s *Spider) parse(ctx context.Context, resp core.IResponseReader) {
    // Use Gosm to extract all product links and pagination in one go
    var list Listing
    _ = gosm.Map(resp, &list)

    // Follow each product link
    for _, productUrl := range list.ProductLinks {
        productUrl = fmt.Sprintf("%s/catalogue/%s", s.baseUrl, productUrl)
        s.Parse(s.Request(ctx).Url(productUrl), s.parseProduct)
    }

    // Follow pagination
    if list.NextPage != "" {
        nextPageUrl := fmt.Sprintf("%s/catalogue/%s", s.baseUrl, list.NextPage)
        s.Parse(s.Request(ctx).Url(nextPageUrl), s.parse)
    }
}

func (s *Spider) parseProduct(ctx context.Context, resp core.IResponseReader) {
    record := &Record{}
    _ = gosm.Map(resp, record)
    s.Yield(record)
}
```

## Record (Gosm Tags)

```go
type Record struct {
    Title       string `gos_css:".product_main h1" json:"title" csv:"title"`
    Price       string `gos_css:".price_color" json:"price" csv:"price"`
    Rating      string `gos_css:".star-rating@class" json:"rating" csv:"rating"`
    Description string `gos_css:"#product_description + *" json:"description" csv:"description"`
    Upc         string `gos_css:"table tr:nth-child(1) td" json:"upc" csv:"upc"`
}

type Listing struct {
    ProductLinks []string `gos_css:"article.product_pod h3 a@href"`
    NextPage     string   `gos_css:"li.next a@href"`
}
```
